\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows, intersections}

\begin{document}
\begin{tikzpicture}
    % Base.
    \draw (0, 0) to (4, 0) ;
    \draw (0, 0) -- (0, 3) ;
    \draw (4, 0) -- (4, 3) ;

    % Horizontal guide line with compass point.
    \draw [gray, dotted] (0, 3) to (2, 3) node[circle, draw, solid] {};
    \draw [gray, dotted] (4, 3) to (2, 3);

    % Vertical guide lines.
    \draw [gray, dotted] (1, 3) to (1, {3+sqrt(3)});
    \draw [gray, dotted] (3, 3) to (3, {3+sqrt(3)});
    % Diagonal guide lines.
    \draw [gray, dotted] (2, 3) to (0, {3+2*sqrt(3)}) node[circle, draw, solid] {};
    \draw [gray, dotted] (2, 3) to (4, {3+2*sqrt(3)}) node[circle, draw, solid] {};
    \draw [gray, dotted] (0, {3+2*sqrt(3)}) to (4, {3+2*sqrt(3)});

    % Arch: lower.
    \draw (0, 3) arc (180:120:2);
    \draw (4, 3) arc (0:60:2);
    % Arch: upper.
    \draw (1, {{3+sqrt(3)}}) arc (-60:0:2);
    \draw (3, {{3+sqrt(3)}}) arc (-120:-180:2);
\end{tikzpicture}
\end{document}
