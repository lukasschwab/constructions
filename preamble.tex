\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows, intersections}

% Define carc: an arc specified by its *center,* (and radius, angle range)
% rather than by its start point.
% 
% Usage example:
% 
%   \draw (0,0) [carc=30:90:1] (0,0) [carc=30:90:2] (0,0) [carc=30:90:3];
% 
% Defines three arcs centered at (0,0), with different radii.
% 
% Source: https://tex.stackexchange.com/a/566645/183126
\tikzset{
  carc/.style args={#1:#2:#3}{
    insert path={+(#1:#3) arc (#1:#2:#3)}
  }
}
