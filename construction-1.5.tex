    \draw [gray, dotted] (0, 5) to (1, 5) node[circle, draw, solid] {} ;

    % Guide lines from base to transition points.
    \draw [gray, dotted] (4, 0) node[circle, draw, solid] {} to (leftTransition) ;
    \draw [gray, dotted] (0, 0) node[circle, draw, solid] {} to (rightTransition) ;

    % Arch: upper.
    \draw [name path=R1] (leftTransition) arc ({180-atan(5/3)}:{180-acos(2/(1+sqrt(34)))}:{1+sqrt(34)}) ;
    \draw [name path=R2] (rightTransition) arc ({atan(5/3)}:{acos(2/(1+sqrt(34)))}:{1+sqrt(34)}) ;

    \draw[name intersections={of=R1 and R2}] (intersection-1);
    \draw [gray, dotted] (4, 0) node[circle, draw, solid] {} to (intersection-1) ;
    \draw [gray, dotted] (0, 0) node[circle, draw, solid] {} to (intersection-1) ;
