\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows, intersections}

\begin{document}
\begin{tikzpicture}
    % Base.
    \draw (0, 0) to (4, 0) ;
    \draw (0, 0) -- (0, 3) ;
    \draw (4, 0) -- (4, 3) ;

    % Compass points.
    \draw [gray, dotted] (2, 3) to (8, 3) node[circle, draw, solid] {};
    \draw [gray, dotted] (2, 3) to (-4, 3) node[circle, draw, solid] {};

    % Arch.
    % The end angles here approximate a gross trigonometric angle while ensuring
    % the arms of the arc intersect.
    \draw [name path=R1] (4, 3) arc (0:41.45:8);
    \draw [name path=R2] (0, 3) arc (180:138.55:8);

    % Compass points to point of arch.
    \draw[name intersections={of=R1 and R2}] (intersection-1);
    \draw [gray, dotted] (8, 3) to (intersection-1);
    \draw [gray, dotted] (-4, 3) to (intersection-1);
\end{tikzpicture}
\end{document}
